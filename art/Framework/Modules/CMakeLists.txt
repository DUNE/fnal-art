  art_Framework_Modules_BlockingPrescaler_module
  art_Framework_Modules_EmptyEvent_source

  art_Framework_Modules_RandomNumberSaver_module

  ARCHIVE DESTINATION ${CMAKE_INSTALL_LIBDIR}
  COMPONENT Runtime
  )
